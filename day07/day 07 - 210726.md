```
#include <stdio.h>

int main() {

    printf("짝수\n\n");

    for(int i = 2; i <= 9; i++) {
        for(int j = 1; j <= 9; j++) {
            if (i % 2 == 0) {
                printf("%d X %d = %d\n",  i, j, i * j);
            }
        }
        printf("-----------------------\n\n");
    }
    printf("홀수\n\n");
    for(int i = 2; i <= 9; i++) {
        for(int j = 1; j <= 9; j++) {
            if (i % 2) {
                printf("%d X %d = %d\n",  i, j, i * j);
            }
        }
        printf("-----------------------\n\n");
    }

    return 0;
}

        printf("-----------------------\n\n");

    printf("홀수\n\n");

            }
        }
        printf("-----------------------\n\n");
    }

    return 0;
}
